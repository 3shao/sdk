143  // Auxiliary method, the actual code of the two front-end functions offered to the user below.
144  template <class LIST_OF_VECTORS1,class LIST_OF_VECTORS2>
145  double stub_kmeans(
146  const bool use_kmeansplusplus_method,
147  const size_t k,
148  const LIST_OF_VECTORS1 & points,
149  std::vector<int> &assignments,
150  LIST_OF_VECTORS2 *out_centers,
151  const size_t attempts
152  )
153  {
154  assert(k>=1);
155 
156  const size_t N = points.size();
157  assignments.resize(N);
158  if (out_centers) out_centers->clear();
159  if (!N)
160  return 0; // No points, we're done.
161  // Parse to required format:
162  size_t dims=0;
163  const typename LIST_OF_VECTORS1::const_iterator it_first=points.begin();
164  const typename LIST_OF_VECTORS1::const_iterator it_end =points.end();
165  typedef typename LIST_OF_VECTORS1::value_type TInnerVector;
166  typedef typename LIST_OF_VECTORS2::value_type TInnerVectorCenters;
167 
168  std::vector<typename TInnerVector::value_type> raw_vals;
169  typename TInnerVector::value_type *trg_ptr=NULL;
170  for (typename LIST_OF_VECTORS1::const_iterator it=it_first;it!=it_end;++it)
171  {
172  if (it==it_first)
173  {
174  dims = it->size();
175  assert(dims>0);
176  raw_vals.resize(N*dims);
177  trg_ptr = &raw_vals[0];
178  }
179  else
180  {
181  assert(size_t(dims)==size_t(it->size()));
182  }
183 
184  ::memcpy(trg_ptr, &(*it)[0], dims*sizeof(typename TInnerVector::value_type));
185  trg_ptr+=dims;
186  }
187  // Call the internal implementation:
188  std::vector<typename TInnerVectorCenters::value_type> centers(dims*k);
189  const double ret = detail::internal_kmeans(false,N,k,points.begin()->size(),&raw_vals[0],attempts,&centers[0],&assignments[0]);
190  // Centers:
191  if (out_centers)
192  {
193  const typename TInnerVectorCenters::value_type *center_ptr = &centers[0];
194  for (size_t i=0;i<k;i++)
195  {
196  TInnerVectorCenters c;
197  for (size_t j=0;j<dims;j++) {
198  if(j < c.size())
199  c[j] = *center_ptr++;
200  }
201  out_centers->push_back(c);
202  }
203  }
204  return ret;
205  }
206  } // end detail namespace
